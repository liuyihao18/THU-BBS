# 数据库

**USER**

```sql
CREATE TABLE USER
(
    userid      interger,
    email       varchar(64) unique,
    nickname    varchar(64),
    password    varchar(128),
    headshot    varchar(256),
    description varchar(512),
    primary key (userid)
)
```

**RELATION**

```sql
CREATE TABLE RELATION
(
    followee integer,
    follower integer,
    foreign key (followee) references USER on delete cascade,
    foreign key (follower) references USER on delete cascade,
    primary key (followee, follower)
)
```

**BLACKLIST**

```sql
CREATE TABLE BLACKLIST
(
    userid  integer, /* 屏蔽的人 */
    blackid integer, /* 被屏蔽的人 */
    primary key (userid, blackid),
    foreign key (userid) references USER on delete cascade,
    foreign key (blackid) references USER on delete cascade
)
```

**TWEET**

```sql
CREATE TABLE TWEET
(
    tweetid       integer,
    userid        integer,
    type          integer,
    isdraft       bool,
    content       varchar(512),
    location      varchar(64),
    lastmodified  datetime,
    like_count    integer,
    comment_count integer,
    primary key (tweetid),
    foreign key (userid) references USER on delete cascade
)
```

**LIKE**

```sql
CREATE TABLE LIKE
(
    tweetid  integer,
    userid   integer,
    likeid   integer,
    liketime datetime,
    primary key (likeid),
    foreign key (tweetid) references TWEET on delete cascade,
    foreign key (userid) references USER on delete cascade
)
```

**PICTURE**

```sql
CREATE TABLE PICTURE
(
    tweetid integer,
    path    varchar(256),
    order   integer, /* 第几张图片 */
    primary key (tweetid, order),
    foreign key (tweetid) references TWEET on delete cascade
)
```

**AUDIO**

```sql
CREATE TABLE AUDIO
(
    tweetid integer,
    path    varchar(256),
    primary key (tweetid),
    foreign key (tweetid) references TWEET on delete cascade
)
```

**VIDEO**

```sql
CREATE TABLE VIDEO
(
    tweetid integer,
    path    varchar(256),
    primary key (tweetid),
    foreign key (tweetid) references TWEET on delete cascade
)
```

**COMMENT**

```sql
CREATE TABLE COMMENT
(
    commentid    integer,
    tweetid      integer,
    userid       integer,
    content      varchar(128),
    comment_time datatime,
    primary key (commentid),
    foreign key (tweetid) references TWEET on delete cascade,
    foreign key (userid) references USER on delete cascade
)
```

**MESSAGE**

```sql
CREATE TABLE MESSAGE
(
    message_id   integer,
    content     varchar(128),
    message_time datetime,
    primary key (messageid)
)
```

**LIKENOTIFICATION**

```sql
CREATE TABLE LIKENOTIFICATION
(
    notification_id integer,
    userid         integer,
    like_user_id    integer,
    like_time       integer,
    primary key (notification_id),
    foreign key (user_id) references USER on delete cascade,
    foreign key (like_user_id) references USER on delete cascade,
)
```

**COMMENTNOTIFICATION**

```sql
CREATE TABLE LIKENOTIFICATION
(
    notification_id integer,
    user_id         integer,
    comment_user_id integer,
    comment_time    datetime,
    comment_id      integer,
    tweet_id 		integer
    primary key (notification_id),
    foreign key (user_id) references USER on delete cascade,
    foreign key (comment_user_id) references USER on delete cascade
)
```
